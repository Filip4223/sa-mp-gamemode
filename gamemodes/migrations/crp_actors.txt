CREATE TABLE `crp_actors` (
  `actor_uid` int(11) NOT NULL AUTO_INCREMENT,
  `actor_type` int(11) NOT NULL,
  `actor_skin` int(11) NOT NULL,
  `actor_name` varchar(40) NOT NULL,
  `actor_px` float NOT NULL,
  `actor_py` float NOT NULL,
  `actor_pz` float NOT NULL,
  `actor_pa` float NOT NULL,
  `actor_anim` varchar(24) NOT NULL,
  `actor_vw` int(11) NOT NULL,
  `actor_int` int(11) NOT NULL,
  PRIMARY KEY (`actor_uid`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
